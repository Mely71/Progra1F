#include <iostream>
using namespace std; 
int main() { 
	
	int numero; 
	cout << "ingrese un numero entero para poder ver si tiene divisor: ";
	
	cin >> numero; 

	for (int entero = 1; entero <= numero; entero++) {
		if (numero % entero)
			cout << "el divisor es: " << entero << "" << endl; 
	}

	
}
